    // Battery status
    function updateBatteryStatus() {
        fetch('http://ari-Xc/topic/power_diagnostic')
            .then(response => response.json())
            .then(data => {
                const power = data.msg;
                document.getElementById('batteryCharge').textContent = Math.round(power.charge);
                document.getElementById('batteryConnected').textContent = power.is_connected ? 'Yes' : 'No';
            })
            .catch(error => console.error('Error fetching battery status:', error));
    }
